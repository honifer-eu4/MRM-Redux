#Joao Takes the Crown
country_event = { 
	id = special_regency_events.1
	title = special_regency_events.1.t
	desc = special_regency_events.1.d
	picture = {
		trigger = {
			has_dlc = "Mandate of Heaven"
		}
		picture = BAD_WITH_MONARCH_eventPicture
	}
	picture = {
		trigger = {
			NOT = { has_dlc = "Art of War" }
		}
		picture = KING_SICK_IN_BED_eventPicture
	}
	
	fire_only_once = yes
	
	trigger = {
		tag = POR
		has_regency = no
		is_lesser_in_union = no
		has_ruler = "Maria I"
		is_year = 1792
		NOT = { is_year = 1816 }
		has_heir = "Joao"
		heir_age = 15
		NOT = { has_country_flag = queen_maria_illness }
		government = monarchy
	}
	
	mean_time_to_happen = {
		months = 24
		modifier = {
			factor = 0.5
			NOT = { stability = 1 }
		}
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = queen_maria_illness
		}
	}
	
	option = { #push him to take the throne
		name = special_regency_events.1.a
		custom_tooltip = special_regency_events.1.tt
		hidden_effect = {
			exile_ruler_as = { name = por_exhiled_maria }
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality = naive_personality }
		}
	}
	
	option = { #she can still rule
		name = special_regency_events.1.b
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -5
		}
		add_estate_loyalty = {
			estate = estate_burghers
			loyalty = -5
		}
	}
}

# Recovering the Château de Chenonceau
country_event = {
	id = special_regency_events.2
	title = special_regency_events.2.t
	desc = special_regency_events.2.d
	picture = GREAT_BUILDING_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = FRA
		is_year = 1519
		NOT = { is_year = 1589 }
		OR = {
			has_ruler = "Catherine"
			has_global_flag = wih_catherine_demedici_flag
		}
		NOT = { has_country_flag = catherine_demedici }
		government = monarchy
	}

	mean_time_to_happen = {
		months = 24
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = catherine_demedici
		}
	}
	
	option = { #we need to end this depravation
		name = special_regency_events.2.a
		if = {
			limit = {
				owns_core_province = 4389
			}
			4389 = {
				if = {
					limit = {
						NOT = { fort_level = 2 }
					}
					add_building = courthouse
				}
				add_province_modifier = {
					name = "chateau_de_chenonceau"
					duration = -1
				}
			}
		}
		else = {
			capital_scope = {
				if = {
					limit = {
						NOT = { fort_level = 2 }
					}
					add_building = courthouse
				}
				add_province_modifier = {
					name = "chateau_de_chenonceau"
					duration = -1
				}
			}
		}
		add_treasury = 50
	}
}

# Sten Sture the Elder
country_event = {
	id = special_regency_events.3
	title = special_regency_events.3.t
	desc = special_regency_events.3.d
	picture = REVOLUTION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = SWE
		has_regency = no
		is_year = 1460
		NOT = { is_year = 1503 }
		NOT = { has_country_flag = sten_sture_regency }
		NOT = { has_ruler = "Sten Gustavsson" }
		government = monarchy
		OR = {
			has_ruler = "Karl VIII"
			NOT = { heir_age = 15 }
			has_heir = no
		}
		is_lesser_in_union = no
		NOT = { dynasty = "Vasa" }
		NOT = { has_ruler_modifier = swe_rise_of_a_new_dynasty }
	}

	mean_time_to_happen = {
		months = 4
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = sten_sture_regency
		}
	}
	
	option = { #he should take care of the country
		name = special_regency_events.3.a
		hidden_effect = {
			exile_ruler_as = { 
				name = swe_karl_exiled
				kill_monarch = no
			}
		}
		if = {
			limit = {
				NOT = { has_ruler = "Sten Gustavsson" }
			}
			if = {
				limit = {
					has_heir = yes
					NOT = { heir_age = 15 }
				}
				define_ruler = {
					name = "Sten Gustavsson"
					dynasty = "Sture"
					regency = yes
					age = 30
					claim = 75
					change_adm = 1
					change_dip = 1
					change_mil = 1
				}
			}
			else = {
				define_ruler = {
					name = "Sten Gustavsson"
					dynasty = "Sture"
					age = 30
					claim = 75
					change_adm = 1
					change_dip = 1
					change_mil = 1
				}
			}
			hidden_effect = {
				add_ruler_personality = scholar_personality
			}
		}
		else = {
			change_mil = 1
			hidden_effect = {
				add_ruler_personality = scholar_personality
			}
		}
		if = {
			limit = {
				exists = DAN
			}
			DAN = {
				add_opinion = {
					who = ROOT
					modifier = sten_sture_opinion
				}
			}
		}
		add_estate_loyalty = {
			estate = estate_burghers
			loyalty = 5
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = -5
		}
		capital_scope = {
			add_unrest = -2
		}
	}
	
	option = { #no
		name = special_regency_events.3.b
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 5
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = 10
		}
		random_owned_province = {
			add_unrest = 3
		}
		random_owned_province = {
			add_unrest = 3
		}
	}
}

# Queen Dowager Yesun Sinyu Persecution
country_event = {
	id = special_regency_events.7
	title = special_regency_events.7.t
	desc = special_regency_events.7.d
	picture = RELIGIOUS_TURMOIL_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		has_regency = no
		is_year = 1759
		NOT = { is_year = 1805 }
		has_country_flag = queen_jung_soon
		any_owned_province = {
			religion_group = christian
		}
	}

	mean_time_to_happen = {
		months = 30
		modifier = {
			factor = 0.8
			estate_loyalty = {
				estate = estate_nobles
				loyalty = 65
			}
		}
	}
	
	option = { #execute the Catholics
		name = special_regency_events.7.a
		every_owned_province = {
			limit = {
				religion_group = christian
			}
			change_religion = ROOT
			add_unrest = 2
		}
	}
	
	option = { #this won't look good on us
		name = special_regency_events.7.b
		every_owned_province = {
			limit = {
				religion_group = christian
			}
			add_unrest = -1
		}
		add_estate_loyalty = {
			estate = estate_nobles 
			loyalty = -10
		}
	}
}
