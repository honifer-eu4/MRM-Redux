						which = moneyAvailable

					}
					check_variable = { which = moneyAvailable which = advisor_base_hiring_cost }
				}
			}
			variable_arithmetic_trigger = {
				custom_tooltip = ""
				export_to_variable = {
					which = moneyAvailable
					value = treasury
				}
				check_variable = { which = moneyAvailable which = advisor_base_hiring_cost }
			}
		}
		else = {
			variable_arithmetic_trigger = {
				custom_tooltip = choice_advisor_nomoney_tooltip

					which = moneyAvailable
					value = treasury
				}
				check_variable = { which = moneyAvailable which = advisor_base_hiring_cost }
			}
		}
	}
	effect = { 
		hidden_effect = {
			set_country_flag = currently_buying_advisor
			ME_calculate_advisor_costs_effect = { skill_level = 3 type = mil }
			hire_advisor = {
				type = grand_captain
				skill = 3
			}
		}
	}
	tooltip = choice_advisor_button_tooltip
