				}

					OR = {
						NOT = { culture = kurdish }

							culture = kurdish

				}

				NOT = { any_province = { culture = kurdish } }
			}

			add_permanent_claim_multi_provinces = { target = mashriq_region }
			custom_tooltip = per_protect_the_west_iranian_tt
			tooltip = {
				add_nationalism = -20
			}
			hidden_effect = {
				every_owned_province = {
					limit = {
						OR = {
							region = anatolia_region
							region = mashriq_region
							region = persia_region
						}
						culture_group = root
					}
					add_nationalism = -20
				}
			}
		}
